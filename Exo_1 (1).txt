// Algorithme qui demande à l'utilisateur un nombre et l'informe ensuite s'il est positif ou négatif 
Algorithme Nombre_positif_ou _négatif 
Variables 
    n : entier ;
Début 
    Écrire ("Veuillez entrer un nombre entier n :");
    Lire (n);
    Si (n>=0) Alors 
        Écrire ("Nombre positif ");
        Sinon 
        Écrire ("Nombre négatif ");
    FinSi 
Fin
